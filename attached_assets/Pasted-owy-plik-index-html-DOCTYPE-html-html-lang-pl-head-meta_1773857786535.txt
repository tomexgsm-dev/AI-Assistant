<script>
const messagesDiv = document.getElementById('messages');
const userInput = document.getElementById('user-input');
const sendBtn = document.getElementById('send-btn');
const clearBtn = document.getElementById('clear-btn');

// Zmień na adres Twojego backendu z Mistral 7B
const backendURL = 'https://TWOJ-BACKEND-URL/chat';

// Funkcja dodawania wiadomości do czatu
function addMessage(text, sender) {
  const msgDiv = document.createElement('div');
  msgDiv.classList.add('message', sender);
  msgDiv.textContent = text;
  messagesDiv.appendChild(msgDiv);
  messagesDiv.scrollTop = messagesDiv.scrollHeight;
}

// Funkcja wysyłania wiadomości
async function sendMessage() {
  const message = userInput.value.trim();
  if (!message) return;
  addMessage(message, 'user');
  userInput.value = '';

  try {
    const response = await fetch(backendURL, {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({message})
    });
    const data = await response.json();
    addMessage(data.reply, 'bot');
  } catch (err) {
    addMessage('Błąd połączenia z backendem', 'bot');
    console.error(err);
  }
}

// Funkcja czyszczenia czatu (frontend)
function clearChat() {
  messagesDiv.innerHTML = '';
}

// Obsługa przycisków i Enter
sendBtn.addEventListener('click', sendMessage);
userInput.addEventListener('keydown', (e) => { if (e.key === 'Enter') sendMessage(); });
clearBtn.addEventListener('click', clearChat);

// Funkcja pobierania historii z backendu (opcjonalnie)
async function loadHistory() {
  try {
    const response = await fetch(`${backendURL}/history`);
    const data = await response.json();
    data.forEach(item => addMessage(item.message, 'user'), addMessage(item.reply, 'bot'));
  } catch (err) {
    console.warn('Nie udało się załadować historii', err);
  }
}

// Jeśli backend obsługuje /history, możesz odkomentować:
