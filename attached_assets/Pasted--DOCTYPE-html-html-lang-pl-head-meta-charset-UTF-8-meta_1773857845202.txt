  <div id="chat-container">
    <h2>Chat Mistral 7B</h2>
    <div id="messages"></div>
    <div id="input-container">
      <input type="text" id="user-input" placeholder="Napisz wiadomość..." />
      <button id="send-btn">Wyślij</button>
    </div>
  </div>

  <script>
    const messagesDiv = document.getElementById('messages');
    const userInput = document.getElementById('user-input');
    const sendBtn = document.getElementById('send-btn');

    const backendURL = 'https://TWOJ-BACKEND-URL/chat'; // <- zmień na adres Twojego backendu z Mistral 7B

    function addMessage(text, sender) {
      const msgDiv = document.createElement('div');
      msgDiv.classList.add('message');
      msgDiv.classList.add(sender);
      msgDiv.textContent = text;
      messagesDiv.appendChild(msgDiv);
      messagesDiv.scrollTop = messagesDiv.scrollHeight;
    }

    async function sendMessage() {
      const message = userInput.value.trim();
      if (!message) return;
      addMessage(message, 'user');
      userInput.value = '';

      try {
        const response = await fetch(backendURL, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ message })
        });

        const data = await response.json();
        addMessage(data.reply, 'bot');
      } catch (err) {
        addMessage('Błąd połączenia z backendem', 'bot');
        console.error(err);
      }
    }

    sendBtn.addEventListener('click', sendMessage);
    userInput.addEventListener('keydown', (e) => {
      if (e.key === 'Enter') sendMessage();
    });
  </script>
